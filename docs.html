Advanced Filtering

üîç Filter by Status Konservasi

JavaScript 
async function filterByStatus(targetStatus, type = 'flora') {
  const response = await fetch('./assets/data/provinsi.json');
  const data = await response.json();
  
  const results = [];
  
  for (const [provinsi, content] of Object.entries(data)) {
    // Check main species
    if (content[type].status.includes(targetStatus)) {
      results.push({ provinsi, type: 'Identitas', ...content[type] });
    }
    
    // Check "lainnya" array
    content[type].lainnya
      .filter(species => species.status === targetStatus)
      .forEach(species => {
        results.push({ provinsi, type: 'Lainnya', ...species });
      });
  }
  
  return results;
}

// Example: Get all critically endangered fauna
const kritisSpecies = await filterByStatus("Kritis", "fauna");

üîé Search by Nama Spesies

JavaScript 
async function searchSpecies(keyword) {
  const response = await fetch('./assets/data/provinsi.json');
  const data = await response.json();
  const searchKey = keyword.toLowerCase();
  const results = [];
  
  for (const [provinsi, content] of Object.entries(data)) {
    // Search in flora
    if (content.flora.nama.toLowerCase().includes(searchKey)) {
      results.push({ provinsi, type: 'Flora Identitas', data: content.flora });
    }
    content.flora.lainnya.forEach(species => {
      if (species.nama.toLowerCase().includes(searchKey)) {
        results.push({ provinsi, type: 'Flora Lainnya', data: species });
      }
    });
    
    // Search in fauna (same pattern)
    if (content.fauna.nama.toLowerCase().includes(searchKey)) {
      results.push({ provinsi, type: 'Fauna Identitas', data: content.fauna });
    }
    content.fauna.lainnya.forEach(species => {
      if (species.nama.toLowerCase().includes(searchKey)) {
        results.push({ provinsi, type: 'Fauna Lainnya', data: species });
      }
    });
  }
  
  return results;
}

// Example: Find all "Harimau" species
const harimauResults = await searchSpecies("harimau");

Framework Integration

Laravel - Service Class 
<?php
// app/Services/BioExploreService.php

namespace App\Services;

use Illuminate\Support\Facades\Http;
use Illuminate\Support\Facades\Cache;

class BioExploreService
{
    private string $apiUrl = 'https://smartonesda.github.io/bioexplore-nusantara/assets/data/provinsi.json';

    /**
     * Get all provinces data (cached for 24 hours)
     */
    public function getAllData(): array
    {
        return Cache::remember('bioexplore_data', 86400, function () {
            $response = Http::get($this->apiUrl);
            return $response->json();
        });
    }

    /**
     * Get data for specific province
     */
    public function getProvinsi(string $nama): ?array
    {
        $data = $this->getAllData();
        return $data[$nama] ?? null;
    }

    /**
     * Get flora identitas
     */
    public function getFloraIdentitas(string $provinsi): ?array
    {
        $data = $this->getProvinsi($provinsi);
        if (!$data) return null;
        
        $flora = $data['flora'];
        unset($flora['lainnya']);
        return $flora;
    }

    /**
     * Get fauna identitas
     */
    public function getFaunaIdentitas(string $provinsi): ?array
    {
        $data = $this->getProvinsi($provinsi);
        if (!$data) return null;
        
        $fauna = $data['fauna'];
        unset($fauna['lainnya']);
        return $fauna;
    }

    /**
     * Search species by name
     */
    public function searchSpecies(string $keyword): array
    {
        $data = $this->getAllData();
        $results = [];
        $keyword = strtolower($keyword);

        foreach ($data as $provinsi => $content) {
            // Search flora
            if (str_contains(strtolower($content['flora']['nama']), $keyword)) {
                $results[] = [
                    'provinsi' => $provinsi,
                    'type' => 'Flora Identitas',
                    'data' => $content['flora']
                ];
            }

            // Search fauna
            if (str_contains(strtolower($content['fauna']['nama']), $keyword)) {
                $results[] = [
                    'provinsi' => $provinsi,
                    'type' => 'Fauna Identitas',
                    'data' => $content['fauna']
                ];
            }
        }

        return $results;
    }

    /**
     * Filter by conservation status
     */
    public function filterByStatus(string $status, string $type = 'flora'): array
    {
        $data = $this->getAllData();
        $results = [];

        foreach ($data as $provinsi => $content) {
            // Check main species
            if (str_contains($content[$type]['status'], $status)) {
                $results[] = [
                    'provinsi' => $provinsi,
                    'type' => ucfirst($type) . ' Identitas',
                    ...$content[$type]
                ];
            }

            // Check lainnya
            foreach ($content[$type]['lainnya'] as $species) {
                if ($species['status'] === $status) {
                    $results[] = [
                        'provinsi' => $provinsi,
                        'type' => ucfirst($type) . ' Lainnya',
                        ...$species
                    ];
                }
            }
        }

        return $results;
    }
}
Laravel - Controller 
<?php
// app/Http/Controllers/BioExploreController.php

namespace App\Http\Controllers;

use App\Services\BioExploreService;
use Illuminate\Http\Request;

class BioExploreController extends Controller
{
    public function __construct(
        private BioExploreService $bioExplore
    ) {}

    public function index()
    {
        $data = $this->bioExplore->getAllData();
        $provinsiList = array_keys($data);
        
        return view('bioexplore.index', compact('provinsiList'));
    }

    public function show(string $provinsi)
    {
        $data = $this->bioExplore->getProvinsi($provinsi);
        
        if (!$data) {
            abort(404, 'Provinsi tidak ditemukan');
        }

        return view('bioexplore.show', [
            'provinsi' => $provinsi,
            'flora' => $data['flora'],
            'fauna' => $data['fauna']
        ]);
    }

    public function search(Request $request)
    {
        $keyword = $request->input('q');
        $results = $this->bioExplore->searchSpecies($keyword);
        
        return response()->json($results);
    }
}
CodeIgniter 4 - Library 
<?php
// app/Libraries/BioExploreLib.php

namespace App\Libraries;

use CodeIgniter\HTTP\CURLRequest;

class BioExploreLib
{
    private string $apiUrl = 'https://smartonesda.github.io/bioexplore-nusantara/assets/data/provinsi.json';
    private $cache;
    private $client;

    public function __construct()
    {
        $this->cache = \Config\Services::cache();
        $this->client = \Config\Services::curlrequest();
    }

    /**
     * Get all data (cached)
     */
    public function getAllData(): array
    {
        $cacheKey = 'bioexplore_all_data';
        
        if (!$data = $this->cache->get($cacheKey)) {
            $response = $this->client->get($this->apiUrl);
            $data = json_decode($response->getBody(), true);
            $this->cache->save($cacheKey, $data, 86400); // 24 hours
        }
        
        return $data;
    }

    /**
     * Get specific province
     */
    public function getProvinsi(string $nama): ?array
    {
        $data = $this->getAllData();
        return $data[$nama] ?? null;
    }

    /**
     * Get flora identitas
     */
    public function getFloraIdentitas(string $provinsi): ?array
    {
        $data = $this->getProvinsi($provinsi);
        if (!$data) return null;
        
        $flora = $data['flora'];
        unset($flora['lainnya']);
        return $flora;
    }

    /**
     * Get fauna identitas
     */
    public function getFaunaIdentitas(string $provinsi): ?array
    {
        $data = $this->getProvinsi($provinsi);
        if (!$data) return null;
        
        $fauna = $data['fauna'];
        unset($fauna['lainnya']);
        return $fauna;
    }

    /**
     * Get all province names
     */
    public function getProvinsiList(): array
    {
        return array_keys($this->getAllData());
    }

    /**
     * Search species
     */
    public function searchSpecies(string $keyword): array
    {
        $data = $this->getAllData();
        $results = [];
        $keyword = strtolower($keyword);

        foreach ($data as $provinsi => $content) {
            if (stripos($content['flora']['nama'], $keyword) !== false) {
                $results[] = [
                    'provinsi' => $provinsi,
                    'type' => 'Flora Identitas',
                    'data' => $content['flora']
                ];
            }
            if (stripos($content['fauna']['nama'], $keyword) !== false) {
                $results[] = [
                    'provinsi' => $provinsi,
                    'type' => 'Fauna Identitas', 
                    'data' => $content['fauna']
                ];
            }
        }

        return $results;
    }
}

<?php
// app/Controllers/BioExplore.php

namespace App\Controllers;

use App\Libraries\BioExploreLib;

class BioExplore extends BaseController
{
    private BioExploreLib $bioExplore;

    public function __construct()
    {
        $this->bioExplore = new BioExploreLib();
    }

    public function index()
    {
        $data = [
            'title' => 'BioExplore Nusantara',
            'provinsiList' => $this->bioExplore->getProvinsiList()
        ];
        
        return view('bioexplore/index', $data);
    }

    public function show(string $provinsi)
    {
        $provinsiData = $this->bioExplore->getProvinsi($provinsi);
        
        if (!$provinsiData) {
            throw \CodeIgniter\Exceptions\PageNotFoundException::forPageNotFound();
        }

        $data = [
            'title' => "Flora & Fauna {$provinsi}",
            'provinsi' => $provinsi,
            'flora' => $provinsiData['flora'],
            'fauna' => $provinsiData['fauna']
        ];
        
        return view('bioexplore/show', $data);
    }

    public function api_search()
    {
        $keyword = $this->request->getGet('q');
        $results = $this->bioExplore->searchSpecies($keyword);
        
        return $this->response->setJSON($results);
    }

    public function api_provinsi(string $nama)
    {
        $data = $this->bioExplore->getProvinsi($nama);
        
        if (!$data) {
            return $this->response->setStatusCode(404)->setJSON([
                'error' => 'Provinsi tidak ditemukan'
            ]);
        }
        
        return $this->response->setJSON($data);
    }
}
React (Custom Hook) 
import { useState, useEffect } from 'react';

function useProvinsiData(namaProvinsi) {
  const [data, setData] = useState(null);
  const [loading, setLoading] = useState(true);
  const [error, setError] = useState(null);

  useEffect(() => {
    async function fetchData() {
      try {
        setLoading(true);
        const res = await fetch('/assets/data/provinsi.json');
        const allData = await res.json();
        setData(allData[namaProvinsi]);
      } catch (err) {
        setError(err.message);
      } finally {
        setLoading(false);
      }
    }
    
    if (namaProvinsi) fetchData();
  }, [namaProvinsi]);

  return { data, loading, error };
}

// Usage in component
function FloraCard({ provinsi }) {
  const { data, loading, error } = useProvinsiData(provinsi);
  
  if (loading) return <div>Loading...</div>;
  if (error) return <div>Error: {error}</div>;
  
  return (
    <div className="flora-card">
      <h2>{data.flora.nama}</h2>
      <p><em>{data.flora.latin}</em></p>
    </div>
  );
}

<script setup>
import { ref, onMounted, watch } from 'vue'

const props = defineProps(['provinsi'])
const flora = ref(null)
const fauna = ref(null)
const loading = ref(true)

async function fetchData() {
  loading.value = true
  try {
    const res = await fetch('/assets/data/provinsi.json')
    const data = await res.json()
    flora.value = data[props.provinsi].flora
    fauna.value = data[props.provinsi].fauna
  } finally {
    loading.value = false
  }
}

onMounted(fetchData)
watch(() => props.provinsi, fetchData)
</script>

<template>
  <div v-if="loading">Loading...</div>
  <div v-else>
    <h2>{{ flora.nama }}</h2>
    <p>{{ flora.latin }}</p>
  </div>
</template>

JavaScript Helper 
function getStatusClass(status) {
  const s = status.toLowerCase();
  if (s.includes('kritis')) return 'status-kritis';
  if (s.includes('terancam')) return 'status-terancam';
  if (s.includes('rentan')) return 'status-rentan';
  if (s.includes('langka')) return 'status-langka';
  if (s.includes('budidaya')) return 'status-budidaya';
  return 'status-umum';
}
